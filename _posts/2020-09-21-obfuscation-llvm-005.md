```c
#include <stdio.h>

int binary(char op, int lhs, int rhs){
	switch (op){
		case '+':
			return lhs + rhs;
		case '-':
			return lhs - rhs;
		case '/':
			return lhs / rhs;
		case '*':
			return lhs * rhs;			
	}
	return lhs;
}

int main(){
	
	int a = 5;
	int b = 6;
	int c = binary('*', a, b);
	
	printf("%d\n", binary('+',a,c));
	
	return 0;
}
```
